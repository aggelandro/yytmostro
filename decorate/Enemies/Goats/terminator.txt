ACTOR DarkTerminator : BaronOfHell2 {
	Health 2000
	Speed 10
	Bloodcolor Green
	PainChance 5
	+BOSS
	+MISSILEMORE
	+MISSILEEVENMORE
	+QUICKTORETALIATE
	SeeSound "cydes/sight" 
  PainSound "cydes/pain"
  DeathSound "cydes/death"
  Obituary "Hasta la vista, bb"
	
	States {
		SpawnLoop:
			DB2S A 14 A_Look
			DB2I A 14 A_Look
			Loop
			
		See:
			TNT1 A 0 A_SetTranslucent(1)
			DB2S AABB 2 A_Chase
			TNT1 A 0 A_PlaySound("commander/metal", CHAN_BODY)
			DB2S CCDD 2 A_Chase
			TNT1 A 0 A_PlaySound("commander/metal", CHAN_BODY)
			TNT1 A 0 A_Jump(62,"Boost")
			Loop
			
		Boost:
			DB2S A 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostA",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			DB2S A 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostA",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			DB2S B 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostB",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			DB2S B 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostB",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			TNT1 A 0 A_PlaySound("commander/metal", CHAN_BODY)
			DB2S C 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostC",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			DB2S C 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostC",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			DB2S D 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostD",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			DB2S D 1 A_Chase
			TNT1 A 0 A_Chase
			TNT1 A 0 A_SpawnItemEx("DarkTerminatorGhostD",0,0,0,0,0,0,0,SXF_CLIENTSIDE|SXF_NOCHECKPOSITION,0)
			TNT1 A 0 A_PlaySound("commander/metal", CHAN_BODY)
			TNT1 A 0 A_Jump(50,"See")
			Loop
			
		Melee:
		Missile:
			DB2S EF 8 A_FaceTarget
			DB2S G 8 A_CustomMissile("BaronBall", 32, 0, frandom(-10, 10))
			Goto See
		Pain:
			DB2S H 2
			DB2S H 2 A_Pain
			TNT1 A -1 A_Jump(256, "See")
			Wait
		Death:
			DB2S I 8
			DB2S J 8 A_Scream
			DB2S K 8
			DB2S L 8 A_NoBlocking
			DB2S MNOP 8
			DB2S Q -1 A_BossDeath
			Stop
		Raise:
			DB2S Q 8
			DB2S PONMLKJI 8
			Goto See
	}
}
