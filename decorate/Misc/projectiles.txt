Actor BulletTracer : AbstractTracer
{
 speed 80
 damage (random(3,4))
 damagetype "Bullet"
 Decal BulletChip
 +BLOODSPLATTER 
 +THRUGHOST
 -DONTSPLASH
 +MISSILE
 States
 {
  Spawn:
   TRAC A 1 NoDelay BRIGHT
   TNT1 A 0 ThrustThingZ(0,random(-8, 8),0,1)
   TRAC A 1 BRIGHT
   Goto Idle	
  Idle:
   TRAC A 1 BRIGHT
   Loop	
  Death:
   TNT1 A 1 A_SpawnItem("BulletPuff")
   Stop
 }
}
